#include <stdio.h>

int main() {
    int sum = 0;
    int product = 1;
    int i, j;
    
    printf("--- For Loop Feature Test ---\n");
    
    // --- Test 1: Simple Increment Loop ---
    // Counts from 0 to 4 (5 iterations)
    // Expected sum: 0+1+2+3+4 = 10
    printf("Test 1: Simple Increment Loop\n");
    for (i = 0; i < 5; i = i + 1) {
        sum = sum + i;
    }
    printf("Final sum: %d (Expected: 10)\n\n", sum);
    
    // --- Test 2: Decrement Loop with Different Start/End ---
    // Counts from 10 down to 7 (10, 9, 8, 7)
    // Expected product: 10 * 9 * 8 * 7 = 5040
    printf("Test 2: Decrement Loop\n");
    for (product = 1, i = 10; i >= 7; i--) {
        product = product * i;
    }
    printf("Final product: %d (Expected: 5040)\n\n", product);
    
    // --- Test 3: Loop with 'break' Statement ---
    // Loop should stop when i reaches 3.
    // sum should be reset, then 0+1+2 = 3
    printf("Test 3: Loop with BREAK\n");
    sum = 0;
    for (i = 0; i < 10; i++) {
        if (i == 3) {
            break; // Exits the loop entirely
        }
        sum = sum + i;
    }
    printf("Final sum after break: %d (Expected: 3)\n\n", sum);
    
    // --- Test 4: Loop with 'continue' Statement ---
    // Loop should skip iteration when i is 2 and 4.
    // sum should be reset, then 0+1+ (skip 2) + 3 + (skip 4) + 5 = 9
    printf("Test 4: Loop with CONTINUE\n");
    sum = 0;
    for (i = 0; i < 6; i++) {
        if (i == 2 || i == 4) {
            continue; // Skips to the next iteration (i++)
        }
        sum = sum + i;
    }
    printf("Final sum after continue: %d (Expected: 9)\n\n", sum);
    
    // --- Test 5: Nested For Loops ---
    // Outer loop (i) runs 2 times (i=0, i=1)
    // Inner loop (j) runs 3 times (j=0, j=1, j=2)
    // Total iterations: 2 * 3 = 6
    printf("Test 5: Nested For Loops\n");
    int count = 0;
    for (i = 0; i < 2; i++) {
        for (j = 0; j < 3; j++) {
            count = count + 1;
        }
    }
    printf("Final count after nesting: %d (Expected: 6)\n\n", count);
    
    // --- Test 6: Loop with Multiple Expressions in Header (Comma Operator) ---
    // Initialization: i=0, j=10
    // Iteration: i++, j--
    // Condition: i < j (stops when i=5, j=5) -> last run when i=4, j=6
    // Expected sum: 0+1+2+3+4 = 10
    printf("Test 6: Multiple Expressions\n");
    sum = 0;
    for (i = 0, j = 10; i < j; i++, j--) {
        sum = sum + i;
    }
    printf("Final sum: %d (Expected: 10)\n", sum);
    printf("Final i: %d (Expected: 5)\n", i);
    
    return 0;
}
