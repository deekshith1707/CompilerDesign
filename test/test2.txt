// for, while, do-while loops with break and continue
#include <stdio.h>

int main() {
    int i, j = 0, k = 0;

    for (i = 0; i < 5; i++) {
        if (i == 2) {
            continue;
        }
        printf("For loop: %d\n", i);
    }

    while (j < 10) {
        if (j == 5) {
            break;
        }
        printf("While loop: %d\n", j);
        j++;
    }

    do {
        printf("Do-while loop: %d\n", k);
        k++;
    } while (k < 3);

    return 0;
}
