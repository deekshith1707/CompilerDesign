//Functions with arguments and recursive calls
#include <stdio.h>

int factorial(int n) {
    if (n == 0) {
        return 1;
    }
    return n * factorial(n - 1);
}

int add(int a, int b) {
    return a + b;
}

int main() {
    int result = add(5, 7);
    printf("Sum: %d\n", result);
    printf("Factorial of 5: %d\n", factorial(5));
    return 0;
}
