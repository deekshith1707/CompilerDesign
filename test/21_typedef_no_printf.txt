// Typedef test without printf
typedef int MyInt;
typedef char MyChar;

typedef struct {
    int a;
    int b;
} MyStruct;

int main() {
    MyInt x = 42;
    MyChar c = 'A';
    
    MyStruct s;
    s.a = 10;
    s.b = 20;
    
    // Just do some operations to verify typedef works
    int sum = s.a + s.b;  // 10 + 20 = 30
    
    return sum;  // Should return 30
}
