// Arithmetic, Logical, if-else, printf, scanf
#include <stdio.h>

int main() {
    int x, y, z;
    printf("Enter two numbers: ");
    scanf("%d %d", &x, &y);

    z = (x * y) + 10;

    if (z > 50 && x != 0) {
        printf("Result is greater than 50.\n");
    } else {
        printf("Result is not greater than 50.\n");
    }
    return 0;
}
