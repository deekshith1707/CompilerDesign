// Test case for for loop
#include <stdio.h>

int main() {
    int i, j, k;
    int arr[10];
    
    // Basic for loop
    for (i = 0; i < 10; i++) {
        arr[i] = i * 2;
    }
    
    for (int x = 0; x < 5; x++) {  
        printf("x = %d\n", x);
    }
    // printf("%d\n", x);            // ERROR: x not in scope outside loop
    
    // Nested for loops
    for (i = 0; i < 3; i++) {
        for ( int x = 0; x < 3; x++) {
            printf("i=%d, x=%d\n", i, x);
        }
    }
    
    // For loop with multiple variables
    for (int x = 0, j = 10; x < j; x++, j--) {
        printf("i=%d, j=%d\n", i, j);
    }
    
    // For loop with different data types
    for (float f = 0.0; f < 10.0; f += 1.5) {
        printf("f = %.1f\n", f);
        // printf("%d", x);  //ERROR
    }
    
    // For loop with pointer arithmetic
    int numbers[5] = {1, 2, 3, 4, 5};
    for (int* ptr = numbers; ptr < numbers + 5; ptr++) {
        printf("Value: %d\n", *ptr);
    }
    
    // Empty for loop components
    i = 0;
    for (;;) {                         // Infinite loop
        if (i >= 5) break;
        printf("i = %d\n", i);
        i++;
    }
    
    // For loop with complex conditions
    for (i = 0; i < 10 && arr[i] < 15; i++) {
        printf("arr[%d] = %d\n", i, arr[i]);
    }
    
    // Scope testing within for loop
    for (i = 0; i < 3; i++) {
        int local_var = i * 10;             // local to this iteration
        printf("Local: %d\n", local_var);
    }
    // printf("%d\n", local_var);          // ERROR: local_var not in scope
    
    for (char* str = "hello"; str; str++) { }  // infinite loop
    for (i = 0; "hello"; i++) { }              // infinite loop
    
    return 0;
}
