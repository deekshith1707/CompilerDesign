// Arrays, pointers, structures
#include <stdio.h>

struct Point {
    int x;
    int y;
};

int main() {
    int arr[3] = {1, 2, 3};
    int *ptr = &arr[1];
    printf("Pointer value: %d\n", *ptr);

    struct Point p;
    p.x = 10;
    p.y = 20;

    struct Point *p_ptr = &p;
    printf("Struct values: %d, %d\n", p_ptr->x, p_ptr->y);

    return 0;
}
