// Test case for printf and scanf - Semantic Analysis
#include <stdio.h>

int main() {
    // Variable declarations for testing
    int i = 42;
    float f = 3.14159;
    char c = 'A';
    char str[100] = "Hello";
    int* ptr = &i;
    
    // Basic printf tests
    printf("Integer: %d\n", i);                     // %d for int
    printf("Float: %f\n", f);                       // %f for float
    printf("Character: %c\n", c);                   // %c for char
    printf("String: %s\n", str);                    // %s for string
    printf("Pointer: %p\n", (void*)ptr);            // %p for pointer
    

    // Multiple arguments
    printf("Multiple: %d %f %c %s\n", i, f, c, str);
    
    // String concatenation in printf
    printf("Line 1\n"
           "Line 2\n"
           "Line 3\n");
    
    // Basic scanf tests
    int input_int;
    float input_float;
    char input_char;
    char input_str[50];
    
    printf("Enter an integer: ");
    scanf("%d", &input_int);                        // %d for int input
    
    printf("Enter a float: ");
    scanf("%f", &input_float);                      // %f for float input
    
    printf("Enter a character: ");
    scanf(" %c", &input_char);                      // Space before %c to skip whitespace
    
    printf("Enter a string: ");
    scanf("%s", input_str);                         // %s for string (no & needed)
    
    // Output the inputs
    printf("You entered: %d, %.2f, %c, %s\n", 
           input_int, input_float, input_char, input_str);
    

    // Multiple inputs in one scanf
    int a, b;
    printf("Enter two integers: ");
    scanf("%d %d", &a, &b);
    
    // Scanf return value checking
    int result = scanf("%d", &input_int);
    if (result == 1) {
        printf("Successfully read one integer\n");
    } else {
        printf("Failed to read integer\n");
    }
    
    return 0;
}
