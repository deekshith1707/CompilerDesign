// Test case for switch-case statements
#include <stdio.h>

int main() {
    int choice = 2;
    char grade = 'B';
    
    // Basic switch statement
    switch (choice) {
        case 1:
            printf("Choice is 1\n");
            break;
        case 2:
            printf("Choice is 2\n");
            break;
        case 3:
            printf("Choice is 3\n");
            break;
        default:
            printf("Invalid choice\n");
            break;
    }
    
    // Switch without break (fall-through)
    switch (choice) {
        case 1:
            printf("Case 1\n");
        case 2:
            printf("Case 2 (or fall-through from 1)\n");
            break;
        case 3:
            printf("Case 3\n");
            break;
    }
    
    // Switch with character
    switch (grade) {
        case 'A':
        case 'a':
            printf("Excellent\n");
            break;
        case 'B':
        case 'b':
            printf("Good\n");
            break;
        case 'C':
        case 'c':
            printf("Average\n");
            break;
        default:
            printf("Invalid grade\n");
    }
    
    // Switch with constants
    int color = 2;
    switch (color) {
        case 1:
            printf("Color is red\n");
            break;
        case 2:
            printf("Color is green\n");
            break;
        case 3:
            printf("Color is blue\n");
            break;
    }
    
    // Nested switch statements
    int category = 1, subcategory = 2;
    switch (category) {
        case 1:
            switch (subcategory) {
                case 1:
                    printf("Category 1, Subcategory 1\n");
                    break;
                case 2:
                    printf("Category 1, Subcategory 2\n");
                    break;
            }
            break;
        case 2:
            printf("Category 2\n");
            break;
    }
    
    /*
    //INVALID CASES 
    int x = 1;
    switch (x) {
        case 1.5:         // ERROR: floating constant not allowed as case label 
            return 0;
    }

    switch (x) {
        case 1:
        case 2:
        case 1:           // ERROR: duplicate case value
            return 0;
    }
    
    const int C = 2;
    switch(x) {
        case C:          //ERROR: In plain C this is NOT an integer constant expression
            return 0;
    }
    */

    return 0;
}
