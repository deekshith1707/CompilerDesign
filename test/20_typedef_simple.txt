#include <stdio.h>

// Simple typedef test
typedef int Integer;
typedef char* String;

typedef struct {
    int x;
    int y;
} Point;

int main() {
    Integer num = 100;
    String msg = "Hello";
    
    Point p;
    p.x = 10;
    p.y = 20;
    
    printf("Number: %d\n", num);
    printf("Message: %s\n", msg);
    printf("Point: (%d, %d)\n", p.x, p.y);
    
    return 0;
}
