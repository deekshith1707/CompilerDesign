// Test Case 6: Functions and Arguments
#include <iostream>
using namespace std;

void printMessage() {
    cout << "Hello" << endl;
}

int add(int a, int b) {
    return a + b;
}

int factorial(int n) {
    if (n <= 1) return 1;
    return n * factorial(n - 1);
}

void staticTest() {
    static int count = 0;
    count++;
}

int main() {
    printMessage();
    int sum = add(5, 10);
    int fact = factorial(5);
    staticTest();
    return 0;
}
