// Test case for arrays (integer and char)
#include <stdio.h>
#include <string.h>

int main() {
    // Integer array declarations
    int arr1[10];                   // Array declaration
    int arr2[5] = {1, 2, 3, 4, 5}; // Array with initialization
    int arr3[] = {10, 20, 30};     // Size inferred from initializer
    int arr4[3] = {1, 2};          // Partial initialization (last element = 0)
    
    // Character arrays
    char str1[20];                  // Character array declaration
    char str2[10] = "hello";        // String initialization
    char str3[] = "world";          // Size inferred from string
    char str4[5] = {'a', 'b', 'c', 'd', '\0'}; // Character by character
    
    // Array access and assignment
    arr1[0] = 100;                  // Valid array access
    arr1[1] = arr2[0] + arr2[1];    // Array element arithmetic
    
    // String operations
    str1[0] = 'h';                  // Character modification
    
    // Array bounds (semantic analyzer should check)
    int i;
    for (i = 0; i < 10; i++) {
        arr1[i] = i * i;            // Valid within bounds
    }
    
    
    // Pointer arithmetic with arrays
    int* ptr = arr1;
    ptr[5] = 999;                   // Equivalent to *(ptr + 5) = 999
    *(arr1 + 3) = 777;             // Direct pointer arithmetic
    
    // Character array operations
    char ch = str2[1];              // Character access
    
    // Type checking scenarios
    printf("arr1[0] = %d\n", arr1[0]);
    printf("str2 = %s\n", str2);    
    
    /*
    // Type errors
    float farr[3.5];             // ERROR: non-integer array size
    int arr5[-5];                // ERROR: negative array size
    arr1 = arr2;                 // ERROR: cannot assign arrays
    int x = arr1;                // ERROR: cannot assign array to int
    arr1[1.5] = 10;              // ERROR: non-integer array index
    arr1["hello"] = 10;          // ERROR: string as array index
    */
    
    return 0;
}
