// Test Case 8: Special Keywords and Control Flow
#include <iostream>
using namespace std;

int main() {
    for (int i = 0; i < 10; i++) {
        if (i == 5) {
            break;
        }
        if (i == 2) {
            continue;
        }
        cout << i << endl;
    }
    
    int j = 0;
    while (true) {
        j++;
        if (j > 10) goto exit;
    }
    
exit:
    cout << "loop exit" << endl;
    
    return 0;
}
