//Dynamic memory allocation and until loop
#include <stdio.h>
#include <stdlib.h>

int main() {
    int *ptr = (int *)malloc(sizeof(int));
    if (ptr == NULL) {
        printf("Memory allocation failed.\n");
        return 1;
    }
    *ptr = 42;
    printf("Dynamically allocated value: %d\n", *ptr);
    free(ptr);

    int counter = 0;
    do {
        printf("Until loop counter: %d\n", counter);
        counter++;
    } until (counter >= 3);

    return 0;
}
