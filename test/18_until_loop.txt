// Until loop
#include <stdio.h>

int is_true() { return 1; }
int is_false() { return 0; }

int main() {
    int x = 0;
    int y = 10;

    // VALID CASES

    x = 0;
    do {
        x++;
    } until (x >= 5);

    x = 0;
    do {
        x++;
    } until (x > 3 && y == 10);

    // 3. Until loop using function call
    x = 0;
    do {
        x++;
    } until (is_true()); 

    // 4. Until loop using break/continue
    x = 0;
    do {
        x++;
        if (x == 3) continue;
        if (x == 5) break;  
    } until (x >= 10);
    
    do {
        int p = 5;
        p++;
    } until(x >= 10);

    // p += 5; // ERROR

    do {
        do{
            int p = 10;
            p--;
        } until( y <= 5);
        // p += 10; // ERROR
    } until(x >= 10);


    /*
    // INVALID CASES

    int arr[5] = {0}; 
    do { 
         x++; 
    } until (arr);     // Warning - Infinite loop

    float f = 1.0;
    int* p = &x;
    do {
      x++;
    } until (f > p); // ERROR : Cannot compare float to pointer

    do {
      x++;
    } until (); // ERROR : Missing expression after 'until'

    until (x >= 10); // ERROR : 'until' keyword used without preceding 'do'
    */
    
    return 0;
}
