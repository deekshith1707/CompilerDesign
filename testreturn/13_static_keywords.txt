                                                                                                                                                                                                                                                    // Test file for static keywords
// Returns 0 on success, or line number on failure

int global_var = 10;
static int file_static_var = 20;

int regular_function(int x) {
    return x + 1;
}

static int static_function(int y) {
    static int local_static = 0;
    local_static = local_static + 1;
    return y + local_static;
}

int counter_function() {
    static int count = 0;
    count = count + 1;
    return count;
}

int main() {
    int result = 0;
    
    // Test 1: Access global variable
    if (global_var != 10) {
        return 30;
    }
    
    // Test 2: Access file static variable
    if (file_static_var != 20) {
        return 35;
    }
    
    // Test 3: Modify global variable
    global_var = 15;
    if (global_var != 15) {
        return 41;
    }
    
    // Test 4: Call regular function
    result = regular_function(5);
    if (result != 6) {
        return 47;
    }
    
    // Test 5: Call static function first time
    result = static_function(10);
    // local_static becomes 1, so 10 + 1 = 11
    if (result != 11) {
        return 54;
    }
    
    // Test 6: Call static function second time
    result = static_function(10);
    // local_static becomes 2, so 10 + 2 = 12
    if (result != 12) {
        return 61;
    }
    
    // Test 7: Call static function third time
    result = static_function(20);
    // local_static becomes 3, so 20 + 3 = 23
    if (result != 23) {
        return 68;
    }
    
    // Test 8: Counter function - first call
    result = counter_function();
    if (result != 1) {
        return 74;
    }
    
    // Test 9: Counter function - second call
    result = counter_function();
    if (result != 2) {
        return 80;
    }
    
    // Test 10: Counter function - third call
    result = counter_function();
    if (result != 3) {
        return 86;
    }
    
    // Test 11: Local static in main
    static int main_static = 40;
    if (main_static != 40) {
        return 92;
    }
    
    // Test 12: Modify local static
    main_static = 50;
    if (main_static != 50) {
        return 98;
    }
    
    return 0; // All tests passed
}
